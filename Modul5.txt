        var result = new int[5];

        for (int i = 0; i < result.Length; i++)
        {
            Console.WriteLine("Введите элемент массива номер {0}", i + 1);

        return result;

    {
        var arrInput = new int[5];

                if (arrInput[i] > arrInput[k])

                    minPrev = arrInput[i];

    {
        var result = new int[5];

            Console.WriteLine("Введите элемент массива номер {0}", i + 1);

        int minPrev = 0;

                {
                    minPrev = result[i];
                    result[i] = result[k];

        return result;
